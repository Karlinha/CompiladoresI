ShellSort

\begin{verbatim}

function shellsort(int vet[], int size) {
    int i , j , value;
    int gap = 1;
    while(gap < size) {
        gap = 3*gap+1;
    }
    while ( gap > 1) {
        gap = gap/3;
        for(i = gap; i < size; i = i + 1) {
            value = vet[i];
            j = i - gap;
            while (j >= 0 and value < vet[j]) {
                vet [j + gap] = vet[j];
                j = j - gap;
            }
            vet [j + gap] = value;
        }
    }
}

\end{verbatim}
